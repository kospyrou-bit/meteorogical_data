final_excel <- read_excel("final_excel.xlsx")

cities <- c("Brasilia", "Rio de Janeiro", "Belo Horizonte", "Salvador",
            "Recife", "Porto Alegre", "São Paulo", "Fortaleza")

max_temp <- final_excel %>%
  filter(City %in% cities) %>%
  group_by(City) %>%
  summarise(Max_Temp = max(Dry_Bulb_Temperature, na.rm = TRUE))

coords <- data.frame(
  City = c("Brasilia","Rio de Janeiro","Belo Horizonte","Salvador",
           "Recife","Porto Alegre","São Paulo","Fortaleza"),
  lon = c(-47.88, -43.20, -43.94, -38.50, -34.88, -51.23, -46.63, -38.52),
  lat = c(-15.79, -22.90, -19.92, -12.97, -8.05, -30.03, -23.55, -3.73)
)

map_data <- left_join(max_temp, coords, by = "City")

brazil <- ne_countries(country = "Brazil", returnclass = "sf")

ggplot() +
  geom_sf(data = brazil, fill = "gray95", color = "black") +
  geom_point(data = map_data,
             aes(x = lon, y = lat, color = Max_Temp),
             size = 5) +
  geom_text(data = map_data,
            aes(x = lon, y = lat, label = City),
            vjust = -1, size = 3) +
  scale_color_gradient(low = "yellow", high = "red") +
  theme_minimal() +
  labs(title = "Maximum Dry Bulb Temperature by City",
       color = "Max Temperature")

#with facet Mean values
mean_twb <- final_excel %>%
  filter(City %in% cities) %>%
  group_by(City, Time_Period) %>%
  summarise(Mean_Twb = mean(Twb, na.rm = TRUE), .groups = "drop")
map_data <- left_join(mean_twb, coords, by = "City")
map_data <- map_data %>%
  mutate(label_vjust = ifelse(City == "São Paulo", 2, -1))
ggplot() +
  geom_sf(data = brazil, fill = "gray95", color = "black") +
  
  # Bigger bubbles
  geom_point(data = map_data,
             aes(x = lon, y = lat, color = Mean_Twb),
             size = 10) +
  
  # Mean value inside bubble
  geom_text(data = map_data,
            aes(x = lon, y = lat, label = round(Mean_Twb,1)),
            color = "black",
            size = 3,
            fontface = "bold") +
  
  # City names (São Paulo below)
  geom_text(data = map_data,
            aes(x = lon, y = lat, label = City, vjust = label_vjust),
            size = 3) +
  
  scale_color_gradient(low = "yellow", high = "red") +
  
  facet_wrap(~Time_Period, ncol = 2) +
  
  theme_minimal() +
  labs(title = "Mean Wet Bulb Temperature (Twb) by City and Time Period",
       color = "Mean Twb")
